120 void
121 CallbackRewardApp::Reward(Ptr<const MobilityModel> observation)
122 {
123  std::vector<float> obs;
124  obs.push_back(observation->GetPosition().x);
125  obs.push_back(observation->GetPosition().y);
126  obs.push_back(observation->GetPosition().z);
127  // reward gets calculated by a function over the length of the vector
128  float length = std::sqrt(std::pow(obs[0], 2) + std::pow(obs[1], 2) + std::pow(obs[2], 2));
129  std::vector<float> reward = {length};
130  Send(CreateDictContainer(reward), 0);
131 }

205  Config::SetDefault("ns3::RandomWalk2dMobilityModel::Time", StringValue("2s"));
206  Config::SetDefault("ns3::RandomWalk2dMobilityModel::Speed",
207  StringValue("ns3::ConstantRandomVariable[Constant=1.0]"));
208  Config::SetDefault("ns3::RandomWalk2dMobilityModel::Bounds", StringValue("0|200|0|200"));
209  CommandLine cmd;

211 
212  MobilityHelper mobility;

214  "X",
215  StringValue("100.0"),

217  StringValue("100.0"),

222  StringValue("Time"),

224  StringValue("2s"),
225  "Speed",
226  StringValue("ns3::ConstantRandomVariable[Constant=1.0]"),
227  "Bounds",

229  mobility.InstallAll();
230 
231  // Create and install observation apps
232  RlApplicationHelper helper(TypeId::LookupByName("ns3::TestObservation"));
233  helper.SetAttribute("StartTime", TimeValue(Seconds(0)));
234  helper.SetAttribute("StopTime", TimeValue(Seconds(10)));
235  RlApplicationContainer observationApps = helper.Install(observationNodes);
236  helper.SetTypeId("ns3::CallbackRewardApp");
237  RlApplicationContainer rewardApps = helper.Install(rewardNodes);
238

241  UintegerValue(10),
242  "MaxRewardHistoryLength",
243  UintegerValue(5));

245

247 
248  commHelper.SetObservationApps(observationApps);
249  commHelper.SetAgentApps(RlApplicationContainer(agentApp));
250  commHelper.SetRewardApps(rewardApps);
251  commHelper.SetIds();
252 
253  NodeContainer internetNodes{observationNodes.Get(0), agentNode};
254  PointToPointHelper p2p;

256  p2p.SetChannelAttribute("Delay", StringValue("2ms"));

259  internet.Install(internetNodes);
260  Ipv4AddressHelper address;
261  address.SetBase("10.1.1.0", "255.255.255.0");
262  Ipv4InterfaceContainer interfaces = address.Assign(internetDevices);
263 
264  commHelper.AddCommunication(
265  {{observationApps.GetId(0),
266  agentApp->GetId(),
267  SocketCommunicationAttributes{interfaces.GetAddress(0), interfaces.GetAddress(1)}},
268  {observationApps.GetId(1), agentApp->GetId(), {}},
269  {rewardApps.GetId(0), agentApp->GetId(), {}}});
270
