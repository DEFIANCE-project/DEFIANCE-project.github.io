93  if (load_checkpoint_path / "best_checkpoint").exists():
94  policies = Policy.from_checkpoint(str(load_checkpoint_path / "best_checkpoint"))
95  else:
96  policies = Policy.from_checkpoint(str(load_checkpoint_path))
97 
98  ns3_settings["visualize"] = ""
99  env = Ns3MultiAgentEnv(targetName=env_name, ns3Path=".", ns3Settings=ns3_settings)
100  reset = env.reset()
101  agent, observation = only(reset[0].items())
102  _, info = only(reset[1].items())
103  terminated, truncated = False, False
104 
105  while True:
106  if "shared_policy" in policies:
107  action = policies["shared_policy"].compute_single_action(observation)[0]
108  else:
109  # this changes based on policy_mapping
110  action = policies[agent].compute_single_action(observation)[0]
111  states = env.step({agent: action})
112 
113  terminated = terminated or states[2]["__all__"]
114  truncated = truncated or states[3]["__all__"]
115 
116  if terminated or truncated:
117  break
118  agent = only(states[0])
119  observation, reward, terminated, truncated, info = (state[agent] for state in states)
120 
121  # Get the termination time if terminated
122  if terminated:
123  agent = only(states[0])
124  observation, reward, terminated, truncated, info = (state[agent] for state in states)
125  logger.info("Average time running: %s", info["terminateTime"])
