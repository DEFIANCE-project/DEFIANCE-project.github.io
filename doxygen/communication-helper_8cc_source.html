66 CommunicationHelper::CreateChannelInterfaces(Ptr<RlApplication> clientApp,

68  const CommunicationAttributes& attributes) const
69 {
70  auto socketAttributes = dynamic_cast<const SocketCommunicationAttributes*>(&attributes);
71  if (!socketAttributes ||
72  (m_checkForSimpleInterfaces && clientApp->GetNode() == serverApp->GetNode()))
73  {
74  auto clientInterface = CreateObject<SimpleChannelInterface>();

78  clientInterface->Connect(serverInterface);

81  return {CreateObject<SocketChannelInterface>(clientApp->GetNode(),
82  socketAttributes->clientAddress,
