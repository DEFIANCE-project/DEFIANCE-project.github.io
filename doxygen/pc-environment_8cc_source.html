176 PowerControlMobComEnv::AddTraffic()
177 {
178  NS_LOG_FUNCTION(this);
179  auto device_manager = GetDeviceManager();
180  Ptr<Node> pgw = m_epcHelper->GetPgwNode();
181 
182  // Create a single RemoteHost
183  NodeContainer remoteHostContainer;
184  remoteHostContainer.Create(1);
185  Ptr<Node> remoteHost = remoteHostContainer.Get(0);
186  InternetStackHelper internet;
187  internet.Install(remoteHostContainer);
188 
189  // Create the Internet
190  PointToPointHelper p2ph;
191  p2ph.SetDeviceAttribute("DataRate", DataRateValue(DataRate("100Gb/s")));
192  p2ph.SetDeviceAttribute("Mtu", UintegerValue(1500));
193  p2ph.SetChannelAttribute("Delay", TimeValue(MilliSeconds(1)));
194  NetDeviceContainer internetDevices = p2ph.Install(pgw, remoteHost);
195  Ipv4AddressHelper ipv4h;
196  ipv4h.SetBase("1.0.0.0", "255.0.0.0");
197  Ipv4InterfaceContainer internetIpIfaces = ipv4h.Assign(internetDevices);
198  Ipv4Address remoteHostAddr = internetIpIfaces.GetAddress(1);
199 
200  Ipv4StaticRoutingHelper ipv4RoutingHelper;
201  Ptr<Ipv4StaticRouting> remoteHostStaticRouting =
202  ipv4RoutingHelper.GetStaticRouting(remoteHost->GetObject<Ipv4>());
203  remoteHostStaticRouting->AddNetworkRouteTo(Ipv4Address("7.0.0.0"), Ipv4Mask("255.0.0.0"), 1);
204 
205  NodeContainer& ueNodes = device_manager->GetUtNodes();
206  NetDeviceContainer& ueDevices = device_manager->GetUtDevices();
207  // Install the IP stack on the UEs
208  internet.Install(ueNodes);
209  Ipv4InterfaceContainer ueIpIface;
210  ueIpIface = m_epcHelper->AssignUeIpv4Address(ueDevices);
211  // Assign IP address to UEs, and install applications
212  for (uint32_t u = 0; u < ueNodes.GetN(); ++u)
213  {
214  Ptr<Node> ueNode = ueNodes.Get(u);
215  // Set the default gateway for the UE
216  Ptr<Ipv4StaticRouting> ueStaticRouting =
217  ipv4RoutingHelper.GetStaticRouting(ueNode->GetObject<Ipv4>());
218  ueStaticRouting->SetDefaultRoute(m_epcHelper->GetUeDefaultGatewayAddress(), 1);
219  }
220 
221  AttachUEsToEnbs();
222 
223  // Install and start applications on UEs and remote host
224  uint16_t dlPort = 1100;
225  uint16_t ulPort = 2000;
226  ApplicationContainer clientApps;
227  ApplicationContainer serverApps;
228  for (uint32_t u = 0; u < ueNodes.GetN(); ++u)
229  {
230  Time interPacketInterval = MilliSeconds(1); // TODO: we could use parameters for this
231 
232  // downlink traffic
233  PacketSinkHelper dlPacketSinkHelper("ns3::UdpSocketFactory",
234  InetSocketAddress(Ipv4Address::GetAny(), dlPort));
235  serverApps.Add(dlPacketSinkHelper.Install(ueNodes.Get(u)));
236 
237  UdpClientHelper dlClient(ueIpIface.GetAddress(u), dlPort);
238  dlClient.SetAttribute("Interval", TimeValue(interPacketInterval));
239  dlClient.SetAttribute("MaxPackets", UintegerValue(1000000));
240  clientApps.Add(dlClient.Install(remoteHost));
241 
242  // uplink traffic
243  ++ulPort;
244  PacketSinkHelper ulPacketSinkHelper("ns3::UdpSocketFactory",
245  InetSocketAddress(Ipv4Address::GetAny(), ulPort));
246  serverApps.Add(ulPacketSinkHelper.Install(remoteHost));
247 
248  UdpClientHelper ulClient(remoteHostAddr, ulPort);
249  ulClient.SetAttribute("Interval", TimeValue(interPacketInterval));
250  ulClient.SetAttribute("MaxPackets", UintegerValue(1000000));
251  clientApps.Add(ulClient.Install(ueNodes.Get(u)));
252  }
253 
254  serverApps.Start(MilliSeconds(100));
255  clientApps.Start(MilliSeconds(100));
256 }
