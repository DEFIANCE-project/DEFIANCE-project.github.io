57 ConnectionStatus
58 SimpleChannelInterface::Connect(Ptr<ChannelInterface> otherInterface)
59 {
60  NS_LOG_FUNCTION(this << otherInterface);
61  try
62  {
63  auto simpleInterface = DynamicCast<SimpleChannelInterface>(otherInterface);
64  if (!simpleInterface)
65  {
66  throw std::bad_cast(); // Checks that we are actually connecting two
67  // SimpleChannelInterfaces with one another
68  }
69 
70  // in case this interface already has a connection - do not connect and keep the old
71  // connection
72  if (GetConnectionStatus() == CONNECTED)
73  {
74  NS_LOG_INFO("Connection failed - this interface is already connected. The old "
75  "connection is kept.");
76  return CONNECTED;
77  }
78 
79  // in case the other interface is already connected to a different interface than this - Do
80  // not allow to connect, the other interface will keep its connection
81  if (simpleInterface->GetConnectionStatus() == CONNECTED &&
82  simpleInterface->GetCommunicationPartner() != this)
83  {
84  NS_LOG_INFO("Connection failed - the other interface is already connected to another "
85  "interface. This interface will stay disconnected.");
86  return DISCONNECTED;
87  }
88 
89  // do not allow to connect to oneself
90  if (simpleInterface == this)
91  {
92  NS_LOG_INFO("Connection failed - connecting to oneself is not allowed.");
93  return DISCONNECTED;
94  }
95 
96  // otherwise connect both interfaces to each other
97  SetCommunicationPartner(simpleInterface);
98  SetConnectionStatus(CONNECTED);
99 
100  simpleInterface->SetCommunicationPartner(this);
101  simpleInterface->SetConnectionStatus(CONNECTED);
102 
103  NS_LOG_INFO("Connecting successful - both interfaces are now connected to each other.");
104  return CONNECTED;
105  }
106  catch (std::bad_cast& e)
107  {
108  NS_LOG_ERROR("CommPartner is not of type SimpleChannelInterface - new connection attempt "
109  "failed. The own connection status now is: "
110  << GetConnectionStatus());
111  return GetConnectionStatus();
112  }
113 }
