28 FindCircleMidpoint(float x1, float x2, float radius)
29 {
30  Point midpoint;
31  float distance = abs(x1 - x2);
32 
33  if (distance > radius * 2)
34  {
35  midpoint.x = 0;
36  midpoint.y = 0;
37  return midpoint;
38  }
39 
40  midpoint.x = (x1 + x2) / 2;
41  midpoint.y = sqrt(pow(radius, 2) - pow(distance, 2));
42 
43  return midpoint;
44 }

103 void
104 EnvironmentCreator::PlaceEnbsRandomly()
105 {
106  srand(43);
107 
108  float from = 0;
109  float to = static_cast<float>(rand()) / (static_cast<float>(RAND_MAX / (m_coverageRadius * 2)));
110 
111  // place the first enb
112  Point circleMiddlePoint = FindCircleMidpoint(from, to, m_coverageRadius);
113  PlaceEnb(m_enbNodes.Get(0), circleMiddlePoint);
114 
115  for (uint32_t i = 1; i < m_enbNodes.GetN(); i++)
116  {
117  // Make the next starting point more likely to be further away
118  from = from + ReweightDiameter(m_coverageRadius * 2);
119  to = from +
120  static_cast<float>(rand()) / (static_cast<float>(RAND_MAX / (m_coverageRadius * 2)));
121 
122  Point circleMiddlePoint = FindCircleMidpoint(from, to, m_coverageRadius);
123  PlaceEnb(m_enbNodes.Get(i), circleMiddlePoint);
124  }
125 
126  m_trackEnd = to; // save how long the overall track is
127 }

166  // for each enb we have one inference agent
167  m_inferenceAgentNodes.Create(numEnbs);
168 
169  m_lteHelper = CreateObject<LteHelper>();
170  m_epcHelper = CreateObject<PointToPointEpcHelper>();
171  m_lteHelper->SetEpcHelper(m_epcHelper);
172 
173  // arguments for the lte helper that are useful for this scenario
174  m_lteHelper->SetAttribute("PathlossModel", StringValue("ns3::FriisPropagationLossModel"));
175  m_lteHelper->SetHandoverAlgorithmType("ns3::A3RsrpHandoverAlgorithm");
176  m_lteHelper->SetHandoverAlgorithmAttribute("Hysteresis", DoubleValue(3.0));
177  m_lteHelper->SetHandoverAlgorithmAttribute("TimeToTrigger", TimeValue(MilliSeconds(256)));
178 
179  InternetStackHelper internet;
180  Ipv4AddressHelper ipv4h;
181  Ipv4StaticRoutingHelper ipv4RoutingHelper;
182 
183  PointToPointHelper p2ph; // TODO: Choose useful arguments
184  p2ph.SetDeviceAttribute("DataRate", DataRateValue(DataRate("100Gb/s")));
185  p2ph.SetDeviceAttribute("Mtu", UintegerValue(1500));
186  p2ph.SetChannelAttribute("Delay", TimeValue(MilliSeconds(10)));
187 
188  Ptr<Node> pgw = m_epcHelper->GetPgwNode();
189 
190  // router
191  // create a router that is connected to the pgw and the inference agent
192  auto router = CreateObject<Node>();
193  NS_LOG_INFO("Node Index of the router: " << router->GetId());
194  internet.Install(router);
195 
196  // p2p link between router and pgw
197  NetDeviceContainer pgwRouterDevices = p2ph.Install(pgw, router);
198 
199  ipv4h.SetBase("192.168.0.0", "255.255.0.0"); // network where the pgw and router live in
200  Ipv4InterfaceContainer pgwRouterInterfaces = ipv4h.Assign(pgwRouterDevices);
201 
202  // Set up routing from router to ue subnet
203  Ptr<Ipv4StaticRouting> routerStaticRouting =
204  ipv4RoutingHelper.GetStaticRouting(router->GetObject<Ipv4>());
205  routerStaticRouting->AddNetworkRouteTo(
206  Ipv4Address("7.0.0.0"),
207  Ipv4Mask("255.0.0.0"),
208  Ipv4Address("192.168.0.1"),
209  1); // 192.168.0.1 is the address of the pgw in this network (next hop)
210 
211  // and also in the other direction from pgw to 1.0.0.0 network
212  Ptr<Ipv4StaticRouting> pgwStaticRouting =
213  ipv4RoutingHelper.GetStaticRouting(pgw->GetObject<Ipv4>());
214  pgwStaticRouting->AddNetworkRouteTo(
215  Ipv4Address("1.0.0.0"),
216  Ipv4Mask("255.0.0.0"),
217  Ipv4Address("192.168.0.2"),
218  3); // 192.168.0.2 is the address of the router which is chosen as the next hop
219 
220  // inference agents
221  internet.Install(m_inferenceAgentNodes);
222  // now we set the ip address for the inference agents (in this case only one)
223  ipv4h.SetBase("1.0.0.0", "255.0.0.0"); // network where router and inference agents live in
224  // now we connect each inference agent to the router
225  for (uint32_t i = 0; i < m_inferenceAgentNodes.GetN(); i++)
226  {
227  NetDeviceContainer routerInferenceAgentDevices =
228  p2ph.Install(router, m_inferenceAgentNodes.Get(i));
229  Ipv4InterfaceContainer routerInferenceAgentInterfaces =

231  NS_LOG_INFO("IP Address of inference agent node "

233  << " ) : " << routerInferenceAgentInterfaces.GetAddress(1));
234  }
235 
236  // we need to set routing from each inference agents to the 7.0.0.0 address via the router
237  // address for the next hop
238  uint16_t j = 1;
239  for (uint32_t u = 0; u < m_inferenceAgentNodes.GetN(); u++)
240  {
241  std::stringstream ss;
242  ss << u + j;

244  ipv4RoutingHelper.GetStaticRouting(m_inferenceAgentNodes.Get(u)->GetObject<Ipv4>());
245  inferenceAgentStaticRouting->AddNetworkRouteTo(Ipv4Address("7.0.0.0"),
246  Ipv4Mask("255.0.0.0"),
247  Ipv4Address(("1.0.0." + ss.str()).c_str()),
248  1);
249  j++;
250  }
251 
252  // and we also set host routing from the router to the inference agents
253  j = 2;
254  for (uint32_t u = 0; u < m_inferenceAgentNodes.GetN(); u++)
255  {
256  std::stringstream ss;
257  ss << u + j; // basically increment by 2 each iteration
258  routerStaticRouting->AddHostRouteTo(Ipv4Address(("1.0.0." + ss.str()).c_str()), j);
259  j++;
260  }
261

263  {

265  // (and vice versa)

267  Ptr<Node> trainingAgentNode = m_trainingAgentNodes.Get(0);
268  internet.Install(trainingAgentNode);
269

271  ipv4h.SetBase("2.0.0.0", "255.0.0.0"); // network where router and training agent live in
272  Ipv4InterfaceContainer routerTrainingAgentInterfaces =
273  ipv4h.Assign(routerTrainingAgentDevices);
274 
275  NS_LOG_INFO("IP Address of training agent node "
276  << 0 << " ( global node index is " << trainingAgentNode->GetId()
277  << " ) : " << routerTrainingAgentInterfaces.GetAddress(1));
278 
279  // now we do routing from the training agent to the inference agents
280  Ptr<Ipv4StaticRouting> trainingAgentStaticRouting =
281  ipv4RoutingHelper.GetStaticRouting(trainingAgentNode->GetObject<Ipv4>());
282  trainingAgentStaticRouting->AddNetworkRouteTo(Ipv4Address("1.0.0.0"),
283  Ipv4Mask("255.0.0.0"),
284  Ipv4Address("2.0.0.1"),
285  1);
286 
287  // and also in the oppsite direction so that inference agent can talk to the training agent
288  routerStaticRouting->AddHostRouteTo(Ipv4Address("2.0.0.0"),
289  numEnbs +
290  2); // first interface connects router and pgw, i+1
291  // connects to the i-th inference agent
292 
293  // we need to set routing from each inference agent to the 2.0.0.0 subnet via the
294  // appropriate router address as next hop
295  j = 1;
296  for (uint32_t u = 0; u < numEnbs; u++)
297  {
298  std::stringstream ss;
299  ss << u + j;
300  Ptr<Ipv4StaticRouting> inferenceAgentStaticRouting =
301  ipv4RoutingHelper.GetStaticRouting(m_inferenceAgentNodes.Get(u)->GetObject<Ipv4>());
302  inferenceAgentStaticRouting->AddNetworkRouteTo(
303  Ipv4Address("2.0.0.0"),
304  Ipv4Mask("255.0.0.0"),
305  Ipv4Address(("1.0.0." + ss.str()).c_str()),
306  1);
307  j++;
308  }
309  }
310 
311  // from now on only setup of ue and enb devices has to follow
312  // Create the networking devices for carts and enb nodes
313  NetDeviceContainer enbDevices;
314  NetDeviceContainer cartDevices;
315 
316  enbDevices = m_lteHelper->InstallEnbDevice(m_enbNodes);
317  cartDevices = m_lteHelper->InstallUeDevice(m_cartNodes);
318 
319  internet.Install(m_cartNodes);
320  Ipv4InterfaceContainer ueIpIface;
321  ueIpIface = m_epcHelper->AssignUeIpv4Address(NetDeviceContainer(cartDevices));
322 
323  for (uint32_t i = 0; i < m_cartNodes.GetN(); i++)
324  {
325  NS_LOG_INFO("IP Address of the cart node "
326  << i << " ( global node index is " << m_cartNodes.Get(i)->GetId() << " ) : "
327  << m_cartNodes.Get(i)->GetObject<Ipv4>()->GetAddress(1, 0).GetAddress());
328  }
329 
330  // Set up routing from the UEs to the PGW
331  for (uint32_t i = 0; i < m_cartNodes.GetN(); ++i)
332  {
333  Ptr<Node> cart = m_cartNodes.Get(i);
334  // Set the default gateway for the carts
335  Ptr<Ipv4StaticRouting> cartStaticRouting =
336  ipv4RoutingHelper.GetStaticRouting(cart->GetObject<Ipv4>());
337  cartStaticRouting->SetDefaultRoute(m_epcHelper->GetUeDefaultGatewayAddress(), 1);
338  }
339 
340  // Attach carts and base stations
341  m_lteHelper->Attach(cartDevices);
342  m_lteHelper->AddX2Interface(m_enbNodes); // we need this enabled for the handover
343 
344  m_lteHelper->GetDownlinkSpectrumChannel()->AddPropagationLossModel(
345  CreateObjectWithAttributes<RangePropagationLossModel>("MaxRange",
346  DoubleValue(m_coverageRadius)));
347 }
