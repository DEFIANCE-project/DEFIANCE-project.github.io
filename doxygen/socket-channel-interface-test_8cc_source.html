55  if (previous == -1.0)

78 SocketChannelInterfaceTestCase::DoRun()
79 {
80  NodeContainer nodes;
81  nodes.Create(2);
82 
83  // Create a point-to-point helper
84  PointToPointHelper p2p;
85  p2p.SetDeviceAttribute("DataRate", StringValue("5Mbps"));
86  p2p.SetChannelAttribute("Delay", StringValue("2ms"));
87 
88  // Create devices and install them on nodes
89  NetDeviceContainer devices;
90  devices.Add(p2p.Install(nodes.Get(0), nodes.Get(1)));
91 
92  // Assign IP addresses
93  InternetStackHelper internet;
94  internet.Install(nodes);
95 
96  Ipv4AddressHelper address;
97  address.SetBase("10.1.1.0", "255.255.255.0");
98  Ipv4InterfaceContainer interfaces = address.Assign(devices);
99 
100  auto udpProtocol = UdpSocketFactory::GetTypeId();
101  auto tcpProtocol = TcpSocketFactory::GetTypeId();
102 
103  auto interfaceSimple0 = CreateObject<SimpleChannelInterface>();
104  auto interfaceSimple1 = CreateObject<SimpleChannelInterface>();
105 
106  auto interfaceUdp0_1A =
107  CreateObject<SocketChannelInterface>(nodes.Get(0), interfaces.GetAddress(0), udpProtocol);
108  auto interfaceUdp0_1B =
109  CreateObject<SocketChannelInterface>(nodes.Get(0), interfaces.GetAddress(0), udpProtocol);
110  auto interfaceUdp1_0 =
111  CreateObject<SocketChannelInterface>(nodes.Get(1), interfaces.GetAddress(1), udpProtocol);
112 
113  auto interfaceTcp0_1A =
114  CreateObject<SocketChannelInterface>(nodes.Get(0), interfaces.GetAddress(0), tcpProtocol);
115  auto interfaceTcp0_1B =
116  CreateObject<SocketChannelInterface>(nodes.Get(0), interfaces.GetAddress(0), tcpProtocol);
117  auto interfaceTcp1_0 =
118  CreateObject<SocketChannelInterface>(nodes.Get(1), interfaces.GetAddress(1), tcpProtocol);
119 
120  // add the receive callback to the channel interfaces
121  interfaceSimple0->AddRecvCallback(
122  MakeCallback(&SocketChannelInterfaceTestCase::ProcessMessage, this, 0));
123  interfaceSimple1->AddRecvCallback(
124  MakeCallback(&SocketChannelInterfaceTestCase::ProcessMessage, this, 1));
125  interfaceUdp0_1A->AddRecvCallback(
126  MakeCallback(&SocketChannelInterfaceTestCase::ProcessMessage, this, 2));
127  interfaceUdp0_1B->AddRecvCallback(
128  MakeCallback(&SocketChannelInterfaceTestCase::ProcessMessage, this, 3));
129  interfaceUdp1_0->AddRecvCallback(
130  MakeCallback(&SocketChannelInterfaceTestCase::ProcessMessage, this, 4));
131  interfaceTcp0_1A->AddRecvCallback(
132  MakeCallback(&SocketChannelInterfaceTestCase::ProcessMessage, this, 5));
133  interfaceTcp0_1B->AddRecvCallback(
134  MakeCallback(&SocketChannelInterfaceTestCase::ProcessMessage, this, 6));
135  interfaceTcp1_0->AddRecvCallback(
136  MakeCallback(&SocketChannelInterfaceTestCase::ProcessMessage, this, 7));
137 
138  Simulator::Schedule(Seconds(0.1),
139  &SimpleChannelInterface::Connect,
140  interfaceSimple0,
141  interfaceSimple1);
142 
143  Simulator::Schedule(Seconds(0.1),
144  &SocketChannelInterface::Connect,
145  interfaceUdp0_1A,
146  interfaceUdp1_0);
147 
148  Simulator::Schedule(Seconds(0.1),
149  &SocketChannelInterface::Connect,
150  interfaceTcp0_1A,
151  interfaceTcp1_0);
152 
153  // Add a simple propagation delay
154  Simulator::Schedule(Seconds(0.3), [&] {
155  interfaceSimple0->SetPropagationDelay(Seconds(0));
156  interfaceSimple1->SetPropagationDelay(Seconds(0));
157  });
158 
159  Simulator::Schedule(Seconds(0.5),
160  &SimpleChannelInterface::Send,
161  interfaceSimple0,
162  MakeDictBoxContainer<float>(1, "box", 1));
163  Simulator::Schedule(Seconds(0.51), [&] { ExpectReceived(1, 1); });
164  Simulator::Schedule(Seconds(0.6),
165  &SocketChannelInterface::Send,
166  interfaceUdp0_1A,
167  MakeDictBoxContainer<float>(1, "box", 4));
168  Simulator::Schedule(Seconds(0.61), [&] { ExpectReceived(4, 4); });
169  Simulator::Schedule(Seconds(0.7),
170  &SocketChannelInterface::Send,
171  interfaceTcp0_1A,
172  MakeDictBoxContainer<float>(1, "box", 7));
173  Simulator::Schedule(Seconds(0.71), [&] { ExpectReceived(7, 7); });
174 
175  // Add a simple propagation delay
176  Simulator::Schedule(Seconds(0.8), [&] { interfaceSimple0->SetPropagationDelay(Seconds(10)); });
177 
178  Simulator::Schedule(Seconds(0.9),
179  &SimpleChannelInterface::Send,
180  interfaceSimple0,
181  MakeDictBoxContainer<float>(1, "box", 42));
182  // Check that message was not received before delay
183  Simulator::Schedule(Seconds(10.89), [&] { ExpectNotReceived(1, 1); });
184  // Check that message was received after delay
185  Simulator::Schedule(Seconds(10.91), [&] { ExpectReceived(1, 42); });
186 
187  // sending in the other direction
188  Simulator::Schedule(Seconds(11),
189  &SimpleChannelInterface::Send,
190  interfaceSimple1,
191  MakeDictBoxContainer<float>(1, "box", 0));
192  Simulator::Schedule(Seconds(11.1), [&] { ExpectReceived(0, 0); });
193 
194  // If we try to reconnect to already connected interfaces this will fail
195  Simulator::Schedule(Seconds(11.1),
196  &SocketChannelInterface::Connect,
197  interfaceUdp0_1B,
198  interfaceUdp1_0);
199  // if we were connected with interfaceUdp0_1B the value 3 would be expected
200  Simulator::Schedule(Seconds(11.2),
201  &SocketChannelInterface::Send,
202  interfaceUdp1_0,
203  MakeDictBoxContainer<float>(1, "box", 2));
204  Simulator::Schedule(Seconds(11.3), [&] { ExpectNotReceived(3, -1); });
205  Simulator::Schedule(Seconds(11.3), [&] { ExpectReceived(2, 2); });
206 
207  // These interfaces are already connected: Therefore, no new connection will be established.
208  // Still, such a call should not lead to a crash or unexpected behaviour.
209  Simulator::Schedule(Seconds(11.3),
210  &SocketChannelInterface::Connect,
211  interfaceTcp1_0,
212  interfaceTcp0_1A);
213  // if we were connected with interfaceTcp0_1B the value 6 would be expected
214  Simulator::Schedule(Seconds(11.4),
215  &SocketChannelInterface::Send,
216  interfaceTcp1_0,
217  MakeDictBoxContainer<float>(1, "box", 5));
218  Simulator::Schedule(Seconds(11.5), [&] { ExpectNotReceived(6, -1); });
219  Simulator::Schedule(Seconds(11.5), [&] { ExpectReceived(5, 5); });
220 
221  // should not be received because the connection is not established
222  Simulator::Schedule(Seconds(11.6),
223  &SocketChannelInterface::Send,
224  interfaceUdp0_1B,
225  MakeDictBoxContainer<float>(1, "box", -2.0));
226  Simulator::Schedule(Seconds(11.65), [&] { ExpectNotReceived(4, 4); });
227  // should not be received because the connection is not established
228  Simulator::Schedule(Seconds(11.7),
229  &SocketChannelInterface::Send,
230  interfaceTcp0_1B,
231  MakeDictBoxContainer<float>(1, "box", -2.0));
232  Simulator::Schedule(Seconds(11.75), [&] { ExpectNotReceived(7, 7); });
233 
234  // but if we closed the old connection before and try again to connect this will work
235  Simulator::Schedule(Seconds(11.8), &SocketChannelInterface::Disconnect, interfaceTcp1_0);
236 
237  // this should not be received
238  Simulator::Schedule(Seconds(11.9),
239  &SocketChannelInterface::Send,
240  interfaceTcp1_0,
241  MakeDictBoxContainer<float>(1, "box", -2.0));
242  Simulator::Schedule(Seconds(11.95), [&] { ExpectNotReceived(5, 5); });
243 
244  // now we do the new connect
245  Simulator::Schedule(Seconds(12),
246  &SocketChannelInterface::Connect,
247  interfaceTcp0_1B,
248  interfaceTcp1_0);
249 
250  // should now be received (used to be connected with interfaceTcp0_1A which would expect 5; but
251  // is now connected with interfaceTcp0_1B)
252  Simulator::Schedule(Seconds(12.1),
253  &SocketChannelInterface::Send,
254  interfaceTcp1_0,
255  MakeDictBoxContainer<float>(1, "box", 6));
256  Simulator::Schedule(Seconds(12.2), [&] { ExpectNotReceived(5, 5); });
257  Simulator::Schedule(Seconds(12.2), [&] { ExpectReceived(6, 6); });
258 
259  // Run the simulation
260  Simulator::Stop(Seconds(15));
261  Simulator::Run();
262  Simulator::Destroy();
263 }
