155 float
156 PowerControlMobComEnv::GetReward()
157 { // negative difference for all ues
158  // under the threshold else inverse
159  // of sum of power
160  NS_LOG_FUNCTION(this);
161 
162  float negativeReward = 0;
163  float powerSum = std::accumulate(powers.begin(), powers.end(), 0);
164  for (auto& cell : sinrs)
165  {
166  for (auto ueSinr : cell)
167  {
168  if (ueSinr < sinrThreshold)
169  {
170  negativeReward += ueSinr - sinrThreshold;
171  }
172  }
173  }
174  float reward = 0.0;
175  if (negativeReward < 0.0)
176  {
177  reward = negativeReward;
178  }
179  else
180  {
181  reward = 1.0 / powerSum;
182  }
183  NS_LOG_INFO("Return Reward " << reward);
184  return reward;
185 }

217  }
