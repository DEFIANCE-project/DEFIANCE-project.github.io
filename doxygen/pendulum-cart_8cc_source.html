138 void
139 PendulumCart::SimulateTimeTic(double deltaT)
140 {
141  NS_LOG_FUNCTION(this << deltaT);
142  // Simulate the physics of the pendulum-cart
143  // based on https://en.wikipedia.org/wiki/Inverted_pendulum
144  // 1. (M + m) x'' - ml theta'' cos(theta) + ml theta'^2 sin(theta) = F
145  // 2. l theta'' - x'' cos(theta) - g sin(theta) = 0
146 
147  // F = (M + m) a
148  // 1. (M + m) x'' - ml theta'' cos(theta) + ml theta'^2 sin(theta) = (M + m) a
149  // 2. l theta'' - x'' cos(theta) - g sin(theta) = 0
150 
151  // 1. x'' = (ml theta'' cos(theta) - ml theta'^2 sin(theta) + (M + m) a) / (M + m)
152  // 2. theta'' = (x'' cos(theta) + g sin(theta)) / l
153 
154  Ptr<ConstantPositionMobilityModel> mobility = this->GetObject<ConstantPositionMobilityModel>();
155 
156  double g = 9.81; // m/s^2
157  double thetaAcceleration =
158  (this->m_acceleration * std::cos(this->m_angle) + g * std::sin(this->m_angle)) /
159  this->m_pendulumLength;
160  this->m_angleVelocity += thetaAcceleration * deltaT;
161 
162  this->SetAngle(this->m_angle + this->m_angleVelocity * deltaT);
163 
164  double cartAcceleration =
165  (this->m_pendulumMass * this->m_pendulumLength * thetaAcceleration *
166  std::cos(this->m_angle) -
167  this->m_pendulumMass * this->m_pendulumLength * this->m_angleVelocity *
168  this->m_angleVelocity * std::sin(this->m_angle) +
169  (this->m_pendulumMass + this->m_cartMass) * this->m_acceleration) /
170  (this->m_pendulumMass + this->m_cartMass);
171 
172  this->m_velocity += cartAcceleration * deltaT;
173  mobility->SetPosition(mobility->GetPosition() + Vector(this->m_velocity * deltaT, 0, 0));
174  this->m_acceleration = this->m_nextAcceleration;
175 
176  // finally we update the trace source so that we have access to the values once they have
177  // changed
178  m_reportCarStatsTrace(GetId(),
179  this->m_cartMass,
180  this->m_pendulumMass,
181  this->m_pendulumLength,
182  this->m_angle,
183  this->m_angleVelocity,
184  mobility->GetPosition(),
185  this->m_velocity,
186  this->m_acceleration,
187  this->m_nextAcceleration);
188 
189  // reschedule the next update
190  Simulator::Schedule(Seconds(deltaT), &PendulumCart::SimulateTimeTic, this, deltaT);
191 }
