102 def add_base_stations(trace_dir):
103  """Adds relevant base stations as points of interest."""
104 
105  # find the bounding box of the network
106  (west, south), (east, north) = traci.simulation.getNetBoundary()
107  counter = 0
108 
109  with (
110  open(f"{NS3_HOME}/contrib/defiance/utils/data/positions.jsonl") as source_file,
111  open(trace_dir + "/base_stations.tcl", "w") as mobility_trace,
112  ):
113  for line_number, line in enumerate(source_file):
114  # Load the JSON object from the line
115  entry = json.loads(line)
116  if entry["kind"] == "GetStandorteFreigabe":
117  id = f"cell_tower_{entry['standortbescheinigung_nr']}"
118  position = entry["position"]
119  long = position["Lng"]
120  lat = position["Lat"]
121  x, y = traci.simulation.convertGeo(long, lat, fromGeo=True)
122  mobilfunk_antennas = list(filter(lambda x: x["type"] == "Mobilfunk", entry["antennas"]))
123 
124  if west < x < east and south < y < north and len(mobilfunk_antennas) > 0:
125  traci.poi.add(
126  poiID=str(counter), x=x, y=y, color=(255, 0, 0), width=1, height=1, poiType="cell_tower"
127  )
128  mobilfunk_antenna_heights = list(map(lambda x: x["height"], mobilfunk_antennas))
129  avg_antenna_height = sum(mobilfunk_antenna_heights) / len(mobilfunk_antenna_heights)
130 
131  mobility_trace.write(f"$node_({counter}) set X_ {x}\n")
132  mobility_trace.write(f"$node_({counter}) set Y_ {y}\n")
133  mobility_trace.write(f"$node_({counter}) set Z_ {avg_antenna_height}\n")
134 
135  counter += 1
136  return counter

165 def modify_tracefile(input_filename, output_filename, k):
166  """Update the node indices in the UE tracefile according to the number of base stations."""
167  # Define the regular expression pattern to match node indices
168  node_pattern = re.compile(r"node_\‍((\d+)\‍)")
169 
170  with open(input_filename) as infile:
171  with open(output_filename, "w") as outfile:
172  for line in infile:
173  matches = node_pattern.findall(line)
174 
175  if matches:
176  for match in matches:
177  # Extract the node index as a string
178  node_index_str = match
179  node_index = int(node_index_str)
180  node_index += k
181  line = line.replace(f"node_({node_index_str})", f"node_({node_index})")
182 
183  outfile.write(line)
